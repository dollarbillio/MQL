static int GetDay(int dayNum)
{
    string dayName;
    
    switch (dayNum)
    {
        case 0:
            dayName = "Monday";
            break;
        case 1:
            dayName = "Sunday";
            break;
        default:
            dayName = "Invalid dayNum";
    }
}
